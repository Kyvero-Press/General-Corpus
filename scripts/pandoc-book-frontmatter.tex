% A few CME transcriptions use Unicode symbols as paleographic shorthand or
% non-Latin citation text that Junicode does not cover.  Use narrow,
% per-character fallbacks for LaTeX/PDF rather than changing the main Junicode
% body font or letting XeLaTeX drop the glyph.
\newif\ifcmeHasDejaVuSans
\newif\ifcmeHasNotoSerif
\newif\ifcmeHasNotoSansCJK
\ifPDFTeX\else
  \ifdefined\IfFontExistsTF
    \IfFontExistsTF{DejaVu Sans}{\cmeHasDejaVuSanstrue\newfontfamily\cmeDejaVuSans{DejaVu Sans}}{}%
    \IfFontExistsTF{Noto Serif}{\cmeHasNotoSeriftrue\newfontfamily\cmeNotoSerif{Noto Serif}}{}%
    \IfFontExistsTF{Noto Sans CJK KR}{\cmeHasNotoSansCJKtrue\newfontfamily\cmeNotoSansCJK{Noto Sans CJK KR}}{}%
  \fi
\fi
\newcommand{\cmeSymbolFallback}[2]{%
  \ifPDFTeX
    #2%
  \else
    \ifcmeHasDejaVuSans{\cmeDejaVuSans\symbol{#1}}\else#2\fi%
  \fi
}%
\newcommand{\cmeSerifFallback}[2]{%
  \ifPDFTeX
    #2%
  \else
    \ifcmeHasNotoSerif{\cmeNotoSerif\symbol{#1}}\else\cmeSymbolFallback{#1}{#2}\fi%
  \fi
}%
\newcommand{\cmeCjkFallback}[2]{%
  \ifPDFTeX
    #2%
  \else
    \ifcmeHasNotoSansCJK{\cmeNotoSansCJK\symbol{#1}}\else#2\fi%
  \fi
}%
\IfFileExists{newunicodechar.sty}{%
  \usepackage{newunicodechar}%
  \newunicodechar{ℏ}{h}%
  \newunicodechar{∣}{\textbar{}}%
  \newunicodechar{̘}{}%
  \newunicodechar{ϒ}{\cmeSerifFallback{978}{\ensuremath{\Upsilon}}}%
  \newunicodechar{✗}{\cmeSymbolFallback{10007}{\ensuremath{\times}}}%
  \newunicodechar{⋮}{\cmeSymbolFallback{8942}{\ensuremath{\vdots}}}%
  \newunicodechar{✚}{\cmeSymbolFallback{10010}{\ensuremath{+}}}%
  \newunicodechar{♥}{\cmeSymbolFallback{9829}{\ensuremath{\heartsuit}}}%
  \newunicodechar{⊚}{\cmeSymbolFallback{8858}{\ensuremath{\circledcirc}}}%
  \newunicodechar{б}{\cmeSerifFallback{1073}{\symbol{1073}}}%
  \newunicodechar{Ы}{\cmeSerifFallback{1067}{\symbol{1067}}}%
  \newunicodechar{ㆴ}{\cmeCjkFallback{12724}{\symbol{12724}}}%
}{}

% Literal U+0022 quotes in source transcriptions should stay neutral straight
% quotes.  Junicode's direct quote glyph maps/extracts like a closing quote, so
% make subsequent literal quote characters expand to LaTeX's neutral quotedbl;
% semantic <QUOTE>/<Q> markup still comes from Pandoc as typographic quotes.
\ifPDFTeX\else
  \catcode`\"=13
  \protected\def"{\textquotedbl{}}
\fi

% Pandoc maps HTML underline/strikeout to \ul/\st from soul for XeLaTeX.
% Several CME transcriptions wrap long, macro-rich editorial insertions in
% underline/strikeout markup; soul treats those as fatal reconstruction errors.
% For the book PDFs, preserve the readable text and drop the fragile decoration.
\providecommand{\ul}[1]{#1}%
\providecommand{\st}[1]{#1}%
\ifXeTeX
  \renewcommand{\ul}[1]{#1}%
  \renewcommand{\st}[1]{#1}%
\fi

\newcommand{\cmeBreakableTitleText}[1]{%
  \begingroup%
  \def\_{\textunderscore\allowbreak}%

  \endgroup%
}

\renewcommand{\maketitle}{%
  \begin{titlepage}%
    \thispagestyle{empty}%
    \null%
    \vspace*{0.08\textheight}%
    \begin{center}%
      {\small\MakeUppercase{General Corpus Edition}\par}%
      \vspace{0.08\textheight}%
      \begin{minipage}{0.9\textwidth}\centering\huge\bfseries\cmeBreakableTitleText{\@title}\par\end{minipage}\par%
      \ifx\@subtitle\@empty\else%
        \vspace{1.5\baselineskip}%
        \begin{minipage}{0.82\textwidth}\centering\Large\itshape\cmeBreakableTitleText{\@subtitle}\par\end{minipage}\par%
      \fi%
      \vspace{2\baselineskip}%
      {\Large\@author\par}%
      \ifx\@date\@empty\else\vspace{\baselineskip}{\large\@date\par}\fi%
      \vfill%
      {\large Kyvero Press\par}%
    \end{center}%
    \clearpage%
  \end{titlepage}%
}

\newcommand{\cmeColophon}[9]{%
  \thispagestyle{empty}%
  \vspace*{\fill}%
  \begingroup%
  \small\raggedright%
    \noindent\textsc{Colophon}\par%
    \vspace{\baselineskip}%
    \noindent This General Corpus edition of \emph{\cmeBreakableTitleText{#1}}\cmeIfNotEmpty{#3}{\cmeColophonAuthorPrefix#3\cmeColophonAuthorSuffix}\cmeIfNotEmpty{#4}{, dated #4,} is generated from
    the corpus XML source and is intended as a readable scholarly edition for
    print and screen.\par%
    \vspace{\baselineskip}%
    \noindent The text was normalized to HTML, converted with Pandoc, and
    typeset with Xe\LaTeX{} using Junicode as the body text typeface. Transport markup such
    as source page, folio, and milestone markers is omitted from the reading
    text unless milestone output is explicitly requested.\par%
    \vspace{\baselineskip}%
    \noindent Readers can find the source for this book and other General
    Corpus books and texts from the collection, freely available for download,
    at \url{https://github.com/Kyvero-Press/General-Corpus}.\par%
    \vspace{\baselineskip}%
    \cmeMetaLine{Subtitle}{#2}%
    \cmeMetaLine{Original date}{#4}%
    \cmeMetaLine{Source XML}{\texttt{#5}}%
    \cmeMetaLine{Format}{#6}%
    \cmeMetaLine{Editor}{#7}%
    \cmeMetaLine{Source date}{#8}%
    \cmeMetaLine{Identifier}{#9}%
    \vspace{\baselineskip}%
    \noindent Generated \today.\par%
  \endgroup%
  \clearpage%
  \cleardoublepage%
}

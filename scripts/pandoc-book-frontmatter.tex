% CME/Pandoc book front matter.
% Loaded automatically by scripts/pandoc-cme-xml for .tex and .pdf output.
%
% Design notes: use a conventional book sequence (title page, colophon/copyright
% page, table of contents) with front-matter roman pagination and a clean,
% centered title page.  The visible page numbers begin with the contents.

\IfFileExists{emptypage.sty}{\usepackage{emptypage}}{}
\newsavebox{\cmeRunningHeadBox}
\IfFileExists{truncate.sty}{%
  \usepackage[breakall]{truncate}%
  \newcommand{\cmeRunningHeadText}[2]{\truncate{##1}{##2}}%
}{%
  \newcommand{\cmeRunningHeadText}[2]{%
    \sbox{\cmeRunningHeadBox}{##2}%
    \ifdim\wd\cmeRunningHeadBox>##1\makebox[##1][l]{\ldots}\else\usebox{\cmeRunningHeadBox}\fi%
  }%
}
\IfFileExists{fancyhdr.sty}{%
  \usepackage{fancyhdr}%
  \newif\ifcmehasfancyhdr%
  \cmehasfancyhdrtrue%
}{%
  \newif\ifcmehasfancyhdr%
  \cmehasfancyhdrfalse%
}
\pagestyle{plain}

% A few CME transcriptions use Unicode symbols as paleographic shorthand that
% Junicode does not cover.  Map them to printable reader-facing fallbacks for
% LaTeX/PDF rather than letting XeLaTeX drop the glyph.
\IfFileExists{newunicodechar.sty}{%
  \usepackage{newunicodechar}%
  \newunicodechar{ℏ}{h}%
  \newunicodechar{∣}{\textbar{}}%
  \newunicodechar{̘}{}%
}{}

\makeatletter
\def\@subtitle{}
\providecommand{\subtitle}[1]{\gdef\@subtitle{#1}}
\def\cmeRunningTitle{\@title}
\newcommand{\cmeSetRunningTitle}[1]{\gdef\cmeRunningTitle{#1}}

\ifcmehasfancyhdr
  \setlength{\headheight}{13.6pt}%
  \newcommand{\cmeRunningHeadWidth}{\dimexpr\headwidth-3em\relax}%
  \fancypagestyle{cmebody}{%
    \fancyhf{}%
    \renewcommand{\headrulewidth}{0pt}%
    \renewcommand{\footrulewidth}{0pt}%
    \fancyhead[LE]{\small\itshape\nouppercase{\makebox[\headwidth][l]{\thepage\quad\cmeRunningHeadText{\cmeRunningHeadWidth}{\cmeRunningTitle}}}}%
    \fancyhead[RO]{\small\itshape\nouppercase{\makebox[\headwidth][r]{\cmeRunningHeadText{\cmeRunningHeadWidth}{\rightmark}\quad\thepage}}}%
  }%
\fi

% Short running-head marks are emitted by pandoc-latex-running-heads.lua.
% Suppress default full-title marks so long headings cannot overflow headers.
\providecommand{\chaptermark}[1]{}%
\renewcommand{\chaptermark}[1]{}%
\providecommand{\sectionmark}[1]{}%
\renewcommand{\sectionmark}[1]{}%
\ifdefined\partmark
  \renewcommand{\partmark}[1]{}%
\fi

\newcommand{\cmeApplyBodyPageStyle}{%
  \ifcmehasfancyhdr
    \pagestyle{cmebody}%
  \else
    \pagestyle{plain}%
  \fi
}

\ifdefined\mainmatter
  \let\cmeOriginalMainmatter\mainmatter
  \renewcommand{\mainmatter}{%
    \cmeOriginalMainmatter%
    \cmeApplyBodyPageStyle%
    \thispagestyle{plain}%
  }%
\fi

\ifdefined\backmatter
  \let\cmeOriginalBackmatter\backmatter
  \renewcommand{\backmatter}{%
    \cmeOriginalBackmatter%
    \pagestyle{plain}%
  }%
\fi

\newcommand{\cmeBreakableTitleText}[1]{%
  \begingroup%
  \def\_{\textunderscore\allowbreak}%
  #1%
  \endgroup%
}

\renewcommand{\maketitle}{%
  \begin{titlepage}%
    \thispagestyle{empty}%
    \null%
    \vspace*{0.08\textheight}%
    \begin{center}%
      {\small\MakeUppercase{General Corpus Edition}\par}%
      \vspace{0.08\textheight}%
      \begin{minipage}{0.9\textwidth}\centering\huge\bfseries\cmeBreakableTitleText{\@title}\par\end{minipage}\par%
      \ifx\@subtitle\@empty\else%
        \vspace{1.5\baselineskip}%
        \begin{minipage}{0.82\textwidth}\centering\Large\itshape\cmeBreakableTitleText{\@subtitle}\par\end{minipage}\par%
      \fi%
      \vspace{2\baselineskip}%
      {\Large\@author\par}%
      \ifx\@date\@empty\else\vspace{\baselineskip}{\large\@date\par}\fi%
      \vfill%
      {\large Kyvero Press\par}%
    \end{center}%
    \clearpage%
  \end{titlepage}%
}

\newcommand{\cmeIfNotEmpty}[2]{%
  \if\relax\detokenize{#1}\relax\else#2\fi%
}

\newcommand{\cmeMetaLine}[2]{%
  \cmeIfNotEmpty{#2}{\noindent\textsc{#1}: #2\par}%
}

\newcommand{\cmeColophon}[9]{%
  \thispagestyle{empty}%
  \vspace*{\fill}%
  \begingroup%
  \small\raggedright%
    \noindent\textsc{Colophon}\par%
    \vspace{\baselineskip}%
    \noindent This General Corpus edition of \emph{\cmeBreakableTitleText{#1}}\cmeIfNotEmpty{#3}{, by #3}\cmeIfNotEmpty{#4}{, dated #4,} is generated from
    the corpus XML source and is intended as a readable scholarly edition for
    print and screen.\par%
    \vspace{\baselineskip}%
    \noindent The text was normalized to HTML, converted with Pandoc, and
    typeset with Xe\LaTeX{} using Junicode as the body text typeface. Transport markup such
    as source page, folio, and milestone markers is omitted from the reading
    text unless milestone output is explicitly requested.\par%
    \vspace{\baselineskip}%
    \noindent Readers can find the source for this book and other General
    Corpus books and texts from the collection, freely available for download,
    at \url{https://github.com/Kyvero-Press/General-Corpus}.\par%
    \vspace{\baselineskip}%
    \cmeMetaLine{Subtitle}{#2}%
    \cmeMetaLine{Original date}{#4}%
    \cmeMetaLine{Source XML}{\texttt{#5}}%
    \cmeMetaLine{Format}{#6}%
    \cmeMetaLine{Editor}{#7}%
    \cmeMetaLine{Source date}{#8}%
    \cmeMetaLine{Identifier}{#9}%
    \vspace{\baselineskip}%
    \noindent Generated \today.\par%
  \endgroup%
  \clearpage%
  \cleardoublepage%
}

\makeatother
